    <script>   
    
    function displayMessage(message, container, isReply = false) {
        const messageDiv = document.createElement('div');
        messageDiv.className = isReply ? 'reply' : 'message';
        
        const isReplayText = message.isReplay ? 'Yanıt' : 'Ana Mesaj';
        const isReplayColor = message.isReplay ? '#dc3545' : '#28a745';
        
        messageDiv.innerHTML = `
            <div class="message-id">ID: ${message.id}</div>
            <div class="message-content">${message.content}</div>
            <div class="message-time">${message.createdAt}</div>
            <div class="message-status" style="color: ${isReplayColor}">${isReplayText}</div>
        `;
        
        // Eğer replay array'i varsa ve boş değilse
        if (message.replay && message.replay.length > 0) {
            const repliesDiv = document.createElement('div');       
            repliesDiv.className = 'replies';
            
            message.replay.forEach(reply => {
                //me Recursive olarak aynı fonksiyonu kullan (isReply = true)
                displayMessage(reply, repliesDiv, true);
            });
            
            messageDiv.appendChild(repliesDiv);
        }
        
        container.appendChild(messageDiv);
    }
    const socket = new SockJS('/ws');
    const stompClient = Stomp.over(socket);

    // Define messagesDiv globally so it's accessible in all scopes
    const messagesDiv = document.getElementById('messages');

    function renderMessages() {
        fetch('/mess')
            .then(response => response.json())
            .then(messages => {
                messagesDiv.innerHTML = '';
                messages.forEach(message => {
                    if (message.isReplay === false) {
                        displayMessage(message, messagesDiv, false);
                    }
                });
            });
    }

    window.addEventListener('DOMContentLoaded', () => {
        renderMessages();

        stompClient.connect({}, () => {
            stompClient.subscribe('/topic/messages', (message) => {
                const msg = JSON.parse(message.body);

                // If it's a new main message, re-render all messages to reflect new state
                // If it's a reply, also re-render all messages to update the thread
                renderMessages();
            });
        });
    });
    
            
        
        
    </script>
